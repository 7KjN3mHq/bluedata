CentOS(4.6/5.2)

修改系统默认语言编码
# vi /etc/sysconfig/i18n
LANG="zh_CN.GBK"
CentOS4.6 再把如下设置改了：
SUPPORTED="zh_CN.GBK:zh_CN:zh"

关闭内核 ipv6 模块
# vi /etc/modprobe.conf
alias net-pf-10 off

CentOS5.2 禁用 SELinux
# vi /etc/sysconfig/selinux
SELINUX=disabled

修改 hosts 文件（可再添加 IP 对应的主机名）
# vi /etc/hosts
127.0.0.1    localhost
CentOS5.2 再注释掉如下设置：
#::1            localhost6.localdomain6 localhost6

配置 yum 源。
# cd /etc/yum.repos.d
# mv CentOS-Media.repo CentOS-Media.repo.bak
# cp -a CentOS-Base.repo CentOS-Base.repo.bak
# vi CentOS-Base.repo
:1,$s/mirrorlist=/#mirrorlist=/g
:1,$s/#baseurl=/baseurl=/g
:1,$s/mirror.centos.org/mirror.be10.com/g
:wq

可以安装使 yum 支持选择较快镜像的补丁
# yum -y install yum-plugin-fastestmirror

安装 vim-enhanced （会自动安装所依赖的 vim-common）
# yum -y install vim-enhanced
配置开启语法着色及关闭显示行号
# vi ~/.vimrc
syntax on
set nohlsearch

常用命令的配置
# vi ~/.bashrc
alias h='history 25'
alias ll='ls -lh --color=tty' 2>/dev/null
alias vi='vim'
# source ~/.bash_profile
# alias -p

使用 ntp 同步时间
CentOS5.2 先：
yum -y install vixie-cron crontabs
# yum -y install ntp
# vi /etc/ntp.conf
server 210.72.145.44

# vi /etc/sysconfig/ntpd
SYNC_HWCLOCK=yes

去掉不需要启动服务
CentOS5.2 先：
# yum -y install ntsysv
# unset LANG
# ntsysv

升级
# yum -y update
# init 6

移除不需要的内核（根据实际情况）
# uname -a
# rpm -qa | grep kernel
# yum -y remove kernel-2.6.9-67.EL kernel-2.6.9-67.0.15.EL kernel-smp-2.6.9-67.EL

安装开发工具
# yum -y install libtool gcc-c++ make
## Dependency Installed: autoconf automake cpp libtool-libs glibc-kernheaders
                         glibc-headers glibc-devel gcc libstdc++-devel

清理下载的包
# yum -e 0 -d 0 clean packages
