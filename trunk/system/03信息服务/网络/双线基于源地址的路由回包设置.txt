$if_tel＝电信IP   $gw_tel＝电信网关
$if_cnc＝网通IP   $gw_cnc＝网通网关

ip route add default via $gw_cnc dev eth1 src $if_cnc table 200
ip rule add from $if_tel table 100
ip rule add from $if_cnc table 200
