在 Ubuntu 上安装 Nginx

http://nginx.org/en/download.html
apt-get update
apt-get install nginx

Nginx 的设置：
cd /etc/nginx
vi nginx.conf
user  www-data;

worker_processes  4;

worker_cpu_affinity  0001 0010 0100 1000;
worker_rlimit_nofile  65536;

error_log  /var/log/nginx/error.log warn;
pid  /var/run/nginx.pid;

events {
    use  epoll;
    worker_connections  65536;
}

http {
    include  mime.types;
    default_type  text/html;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    log_format  custom  '$remote_addr [$time_local] $host "$request_uri" '
                        '$status $request_time $upstream_response_time "$http_user_agent"';

    access_log  /var/log/nginx/access.log  custom;

    server_tokens  off;

    sendfile        on;
    tcp_nopush      on;
    tcp_nodelay     on;

    keepalive_timeout  0;

    gzip  on;
    gzip_min_length  1100;
    gzip_buffers  4 8k;
    gzip_types  text/css text/xml application/x-javascript text/plain;
    gzip_comp_level  9;
    gzip_proxied  any;
    gzip_vary  on;
    gzip_http_version  1.0;
    gzip_disable  "msie6";

    output_buffers  4 32k;
    postpone_output  1460;

    client_header_buffer_size  128k;
    large_client_header_buffers  4 256k;
    client_header_timeout  1m;
    client_body_timeout    1m;
    client_max_body_size   8m;
    send_timeout           1m;

    include fastcgi.conf;
    include /etc/nginx/conf.d/*.conf;
}

echo 'fastcgi_connect_timeout 300s;
fastcgi_send_timeout 300s;
fastcgi_read_timeout 300s;
fastcgi_buffer_size 128k;
fastcgi_buffers 8 128k;
fastcgi_busy_buffers_size 256k;
fastcgi_temp_file_write_size 256k;' > fastcgi.conf

vi conf.d/default.conf
server {
    listen  80;
    server_name  _;

    set $webroot $host;

    if ( $host ~* www\.(.*) ) {
        set $webroot $1;
    }

    location / {
        root  /home/www/$webroot;
        index  index.html index.htm index.php;
    }

    location ~* ^.+.(html|htm)$ {
        root  /home/www/$webroot;
        expires  1h;
    }

    location ~* ^.+.(jpg|jpeg|gif|png|ico|css|zip|tgz|gz|rar|bz2|doc|xls|exe|pdf|ppt|txt|tar|mid|midi|wav|bmp|rtf|swf|js|asp)$ {
        root  /home/www/$webroot;
        access_log  off;
        expires  30d;
    }
 
    location ~ \.php$ {
        fastcgi_pass  127.0.0.1:9000;
        fastcgi_param  SCRIPT_FILENAME /home/www/$webroot/$fastcgi_script_name;
        include  fastcgi_params;
    }

    location ~ /\.ht {
        deny  all;
    }

    location /nginx_status {
        stub_status  on;
        access_log  off;
        allow  127.0.0.1;
        deny  all;
    }
}

子域名对应域名子目录的配置：
server {
    listen  80;
    server_name  *.domain.com domain.com;

    if ( $host ~* ^domain\.com$ ) {
        set $webroot domain.com/www;
    }

    if ( $host ~* (.*)\.domain\.com ) {
        set $webroot domain.com/$1;
    }

    include  location.conf;
}

系统的相关设置：
echo '
fs.file-max = 65536
' >> /etc/sysctl.conf
sysctl -p

echo '
www-data soft nofile 65536
www-data hard nofile 65536
' >> /etc/security/limits.conf

ulimit -SHn 65536
service nginx restart
