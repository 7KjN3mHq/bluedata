mkdir /redis
cd /redis
mkdir bin data etc log run sbin swap
groupadd redis
useradd -g redis -d /redis -s /bin/bash redis

wget http://redis.googlecode.com/files/redis-2.2.4.tar.gz
tar zxvf redis-2.2.4.tar.gz
cd redis-2.2.4

make
cp redis.conf /redis/etc/
cd src
cp redis-benchmark redis-check-aof redis-check-dump redis-cli redis-server /redis/bin/

echo '#! /bin/sh
### BEGIN INIT INFO
# Provides:		redis-server
# Required-Start:	$syslog $remote_fs
# Required-Stop:	$syslog $remote_fs
# Should-Start:		$local_fs
# Should-Stop:		$local_fs
# Default-Start:	2 3 4 5
# Default-Stop:		0 1 6
# Short-Description:	redis-server - Persistent key-value db
# Description:		redis-server - Persistent key-value db
### END INIT INFO


PATH=/usr/local/sbin:/usr/local/bin:/sbin:/bin:/usr/sbin:/usr/bin
DAEMON=/redis/bin/redis-server
DAEMON_ARGS=/redis/etc/redis.conf
NAME=redis-server
DESC=redis-server
PIDFILE=/var/run/redis.pid

test -x $DAEMON || exit 0

set -e

case "$1" in
  start)
	echo -n "Starting $DESC: "
	touch $PIDFILE
	chown redis:redis $PIDFILE
	if start-stop-daemon --start --quiet --umask 007 --pidfile $PIDFILE --chuid redis:redis --exec $DAEMON -- $DAEMON_ARGS
	then
		echo "$NAME."
	else
		echo "failed"
	fi
	;;
  stop)
	echo -n "Stopping $DESC: "
	if start-stop-daemon --stop --retry 10 --quiet --oknodo --pidfile $PIDFILE --exec $DAEMON
	then
		echo "$NAME."
	else
		echo "failed"
	fi
	rm -f $PIDFILE
	;;

  restart|force-reload)
	${0} stop
	${0} start
	;;

  status)
	echo -n "$DESC is "
	if start-stop-daemon --stop --quiet --signal 0 --name ${NAME} --pidfile ${PIDFILE}
	then
		echo "running"
	else
		echo "not running"
		exit 1
	fi
	;;

  *)
	echo "Usage: /etc/init.d/$NAME {start|stop|restart|force-reload}" >&2
	exit 1
	;;
esac

exit 0' > /redis/sbin/redis-server

chmod +x /redis/sbin/redis-server

echo '
vm.overcommit_memory = 1' >> /etc/sysctl.conf

mv /redis/etc/redis.conf /redis/etc/redis.conf.default

echo "daemonize yes

pidfile /redis/run/redis.pid

port 6379

#bind $EC2_LOCAL_IPV4

timeout 300

loglevel notice

logfile  /redis/log/redis.log

databases 16

save 900 1
save 300 10
save 60 10000
save 30 100000

rdbcompression yes

dbfilename database.rdb

dir /redis/data

appendonly no

appendfsync everysec

vm-enabled no

vm-swap-file /redis/swap/redis.swap

vm-max-memory 0

vm-page-size 32

vm-pages 134217728

vm-max-threads 4

glueoutputbuf yes

hash-max-zipmap-entries 64
hash-max-zipmap-value 512

activerehashing yes" > /redis/etc/redis.conf

chown -R redis:redis /redis

sysctl vm.overcommit_memory=1
/redis/sbin/redis-server start
