## Install mysql-server
apt-get install mysql-server
The following NEW packages will be installed:
  bsd-mailx exim4 exim4-base exim4-config exim4-daemon-light libdbd-mysql-perl libdbi-perl libhtml-template-perl liblockfile1
  libmysqlclient15off libnet-daemon-perl libpcre3 libplrpc-perl mailx mysql-client-5.0 mysql-common mysql-server mysql-server-5.0


## Configure
echo -ne "
fs.file-max = 65536
" >> /etc/sysctl.conf
sysctl -p

echo -ne "
mysql soft nofile 65536
mysql hard nofile 65536
" >> /etc/security/limits.conf

vi /etc/mysql/my.cnf
datadir         = /mysql
skip-name-resolve

log_slow_queries        = /var/log/mysql/mysql-slow.log
long_query_time = 1
log-queries-not-using-indexes

skip-innodb


chown -R mysql:mysql /mysql
/etc/init.d/mysql stop
mv /var/lib/mysql/mysql /mysql/
rm -rf /var/lib/mysql/*
/etc/init.d/mysql start
